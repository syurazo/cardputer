1//! Initialize I2C driver for Grove I/F
2use anyhow::Result;
3use esp_idf_hal::{
4    gpio::{Gpio1, Gpio2},
5    i2c::I2C0,
6    i2c::{I2cConfig, I2cDriver},
7    peripheral::Peripheral,
8    units::Hertz,
9};
10
11pub fn build<'a>(
12    i2c: impl Peripheral<P = I2C0> + 'a,
13    sda: impl Peripheral<P = Gpio2> + 'a,
14    scl: impl Peripheral<P = Gpio1> + 'a,
15    hz: Hertz,
16) -> Result<I2cDriver<'a>> {
17    let config = I2cConfig::new().baudrate(hz);
18    let i2c = I2cDriver::new(i2c, sda, scl, &config)?;
19
20    Ok(i2c)
21}
