104#[derive(Debug, PartialEq, Clone, Copy)]
105/// Conversion rule
106pub struct ConversionRule(KeyImprint, Modified, Modified);
107impl ConversionRule {
108    /// Convert according to the state of Fn and Shift key
109    pub fn modified(&self, is_fn_pressed: bool, is_shift_pressed: bool) -> Modified {
110        match (self.0, is_fn_pressed, is_shift_pressed) {
111            (KeyImprint::SemiColon, true, _) => Modified::UpCursor,
112            (KeyImprint::Period, true, _) => Modified::DownCursor,
113            (KeyImprint::Slash, true, _) => Modified::RightCursor,
114            (KeyImprint::Comma, true, _) => Modified::LeftCursor,
115            (KeyImprint::Backquote, true, _) => Modified::Escape,
116            (KeyImprint::Backspace, true, _) => Modified::Delete,
117            (_, _, true) => self.2,
118            (_, _, _) => self.1,
119        }
120    }
121
122    /// Returns the imprint of the key assigned to the rule
123    pub fn imprint(&self) -> KeyImprint {
124        self.0
125    }
126}

231/// Keyboard scanner for Cardputer
232///
233/// # Examples
234///
235/// ```
236/// use cardputer::keyboard::{Keyboard, KeyImprint};
237///
238/// let peripherals = Peripherals::take().unwrap();
239///
240/// let mut keyboard = Keyboard::new(
241///     peripherals.pins.gpio8,
242///     peripherals.pins.gpio9,
243///     peripherals.pins.gpio11,
244///     peripherals.pins.gpio13,
245///     peripherals.pins.gpio15,
246///     peripherals.pins.gpio3,
247///     peripherals.pins.gpio4,
248///     peripherals.pins.gpio5,
249///     peripherals.pins.gpio6,
250///     peripherals.pins.gpio7,
251/// )
252/// .unwrap();
253/// let keys: Vec<KeyImprint> = keyboard.scan_pressed_keys().unwrap();
254/// ```
255pub struct Keyboard<'a> {
256    addr: [PinDriver<'a, AnyOutputPin, Output>; 3],
257    inputs: [PinDriver<'a, AnyIOPin, Input>; 7],
258}
259impl<'a> Keyboard<'a> {
260    /// Create new scanner.
261    #[allow(clippy::too_many_arguments)]
262    pub fn new(
263        a0: impl Peripheral<P = Gpio8> + 'a + OutputPin,
264        a1: impl Peripheral<P = Gpio9> + 'a + OutputPin,
265        a2: impl Peripheral<P = Gpio11> + 'a + OutputPin,
266        y0: impl Peripheral<P = Gpio13> + 'a + IOPin,
267        y1: impl Peripheral<P = Gpio15> + 'a + IOPin,
268        y2: impl Peripheral<P = Gpio3> + 'a + IOPin,
269        y3: impl Peripheral<P = Gpio4> + 'a + IOPin,
270        y4: impl Peripheral<P = Gpio5> + 'a + IOPin,
271        y5: impl Peripheral<P = Gpio6> + 'a + IOPin,
272        y6: impl Peripheral<P = Gpio7> + 'a + IOPin,
273    ) -> Result<Self> {
274        let addr = [
275            PinDriver::output(a0.downgrade_output())?,
276            PinDriver::output(a1.downgrade_output())?,
277            PinDriver::output(a2.downgrade_output())?,
278        ];
279        let mut inputs = [
280            PinDriver::input(y0.downgrade())?,
281            PinDriver::input(y1.downgrade())?,
282            PinDriver::input(y2.downgrade())?,
283            PinDriver::input(y3.downgrade())?,
284            PinDriver::input(y4.downgrade())?,
285            PinDriver::input(y5.downgrade())?,
286            PinDriver::input(y6.downgrade())?,
287        ];
288        for pin in inputs.iter_mut() {
289            pin.set_pull(Pull::Up)?;
290        }
291        Ok(Self { addr, inputs })
292    }
293}

295impl KeyboardScanner for Keyboard<'_> {
296    fn scan_pressed_keys(&mut self) -> Result<Vec<KeyImprint>> {
297        let mut keys: Vec<KeyImprint> = vec![];
298        for i in 0..8 {
299            for (j, ad) in self.addr.iter_mut().enumerate() {
300                ad.set_level(pin_level!(i & (0b00000001 << j)))?;
301            }
302            let inputs: Vec<Level> = self.inputs.iter().map(|x| x.get_level()).collect();
303            for (j, decoded) in inputs.iter().enumerate() {
304                if *decoded == Level::High {
305                    continue;
306                }
307                let (col, row) = if i < 4 {
308                    (COLUMN_MAP[0][j], i)
309                } else {
310                    (COLUMN_MAP[1][j], i - 4)
311                };
312                keys.push(KEY_MAP[row][col]);
313            }
314        }
315
316        Ok(keys)
317    }
318}

320/// Structure that scans the keyboard and keeps track of state changes
321///
322/// # Examples
323///
324/// ```
325/// use cardputer::keyboard::{Keyboard, KeyboardState};
326///
327/// let peripherals = Peripherals::take().unwrap();
328///
329/// let mut keyboard = Keyboard::new(
330///     peripherals.pins.gpio8,
331///     peripherals.pins.gpio9,
332///     peripherals.pins.gpio11,
333///     peripherals.pins.gpio13,
334///     peripherals.pins.gpio15,
335///     peripherals.pins.gpio3,
336///     peripherals.pins.gpio4,
337///     peripherals.pins.gpio5,
338///     peripherals.pins.gpio6,
339///     peripherals.pins.gpio7,
340/// )
341/// .unwrap();
342///
343/// let mut keyboard_state = KeyboardState::default();
344/// keyboard_state.update(&mut keyboard).unwrap();
345/// log::info!("{:?}", keyboard_state.pressed_keys());
346/// log::info!("{:?}", keyboard_state.released_keys());
347/// ```
348#[derive(Debug, Default)]
349pub struct KeyboardState {
350    is_fn_pressed: bool,
351    is_ctrl_pressed: bool,
352    is_shift_pressed: bool,
353    is_alt_pressed: bool,
354
355    hold_keys: Vec<ConversionRule>,
356    pressed_keys: Vec<ConversionRule>,
357    released_keys: Vec<ConversionRule>,
358}
359
360impl KeyboardState {
361    /// Get the latest key state and update the Pressed/Released state
362    pub fn update(&mut self, keyboard: &mut impl KeyboardScanner) -> Result<()> {
363        let mut new_hold_keys: Vec<ConversionRule> = Vec::new();
364
365        self.pressed_keys.clear();
366        self.released_keys.clear();
367
368        self.is_fn_pressed = false;
369        self.is_ctrl_pressed = false;
370        self.is_shift_pressed = false;
371        self.is_alt_pressed = false;
372
373        for pressed in keyboard.scan_pressed_keys()?.into_iter() {
374            let key_type: KeyType = pressed.into();
375            match key_type {
376                KeyType::Modifier(KeyImprint::LeftFn) => self.is_fn_pressed = true,
377                KeyType::Modifier(KeyImprint::LeftCtrl) => self.is_ctrl_pressed = true,
378                KeyType::Modifier(KeyImprint::LeftShift) => self.is_shift_pressed = true,
379                KeyType::Modifier(KeyImprint::LeftAlt) => self.is_alt_pressed = true,
380                KeyType::Normal(h) => {
381                    new_hold_keys.push(h);
382                    if !self.hold_keys.contains(&h) {
383                        self.pressed_keys.push(h);
384                    }
385                }
386                _ => {}
387            }
388        }
389
390        for key in self.hold_keys.iter() {
391            if !new_hold_keys.contains(key) {
392                self.released_keys.push(*key);
393            }
394        }
395
396        self.hold_keys = new_hold_keys;
397
398        Ok(())
399    }
400
401    pub fn pressed_keys(&self) -> Vec<Modified> {
402        self.pressed_keys
403            .iter()
404            .map(|x| x.modified(self.is_fn_pressed, self.is_shift_pressed))
405            .collect()
406    }
407
408    pub fn released_keys(&self) -> Vec<Modified> {
409        self.released_keys
410            .iter()
411            .map(|x| x.modified(self.is_fn_pressed, self.is_shift_pressed))
412            .collect()
413    }
414
415    pub fn hold_keys(&self) -> Vec<Modified> {
416        self.hold_keys
417            .iter()
418            .map(|x| x.modified(self.is_fn_pressed, self.is_shift_pressed))
419            .collect()
420    }
421
422    pub fn is_fn_pressed(&self) -> bool {
423        self.is_fn_pressed
424    }
425
426    pub fn is_ctrl_pressed(&self) -> bool {
427        self.is_ctrl_pressed
428    }
429
430    pub fn is_shift_pressed(&self) -> bool {
431        self.is_shift_pressed
432    }
433
434    pub fn is_alt_pressed(&self) -> bool {
435        self.is_alt_pressed
436    }
437}
